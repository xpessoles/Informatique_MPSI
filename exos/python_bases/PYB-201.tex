\question{On considère les affectations \pyv{a} $\leftarrow$ \pyv{-1.} et \pyv{b} $\leftarrow$ \pyv{5}.
Prévoir la valeur de chacune de ces expressions, puis le vérifier à l'aide de l'interpréteur interactif.}

\begin{multicols}{2}
  \begin{enumerate}[label=\emph{\alph*)}]
    \item \pyv{a * a}
    \item \pyv{a ** a}
    \item \pyv{a ** a == a}
    \item \pyv{a * b}
    \item \pyv{a+b == 5}
    \item \pyv{a+6>=b}
    \item \pyv{b<100 | a**2==-1}
    \item \pyv{b-3}
  \end{enumerate}
\end{multicols}
