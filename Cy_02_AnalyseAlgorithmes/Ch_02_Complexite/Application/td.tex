%O(n racine(n))
\begin{pyverbatim}
def f3(n):
    x = 0
    for i in range(n):
        j = 0
        while j * j < i:
            x += 1
            j += 1
    return x
\end{pyverbatim}


% O log(n)
\begin{pyverbatim}
def f4(n):
    x, i = 0, n
    while i > 1:
        x += 1
        i //= 2
    return x
\end{pyverbatim}
